    public List<List<Integer>> threeSum(int[] nums) {

 * Definition for singly-linked list.
 * class ListNode {
 *     int val;

 *         next = null;
 *     }

public class Solution {
    public boolean hasCycle(ListNode head) {
        if (head == null) {
            return false;
        }
        ListNode pre = head;
        ListNode next = head;
        while (next != null) {
            pre = pre.next;
            next = next.next;
            if (next != null) {
                next = next.next;
            } else {
                return false;
            }
            if (pre == next) {
                return true;
            }
        }
        return false;
    }
}
